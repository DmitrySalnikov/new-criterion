  T_n(Z) = &-&\frac{1}{n^2}\sum_{i,j=1}^n\ln(1+(X_i-Y_j)^2)+ \\
           &+&\frac{1}{n(n-1)}\sum_{i<j}^n\ln(1+(X_i-X_j)^2)+\frac{1}{n(n-1)}\sum_{i<j}^n\ln(1+(Y_i-Y_j)^2).
\end{eqnarray*}
