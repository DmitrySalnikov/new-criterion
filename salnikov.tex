\begin{eqnarray}
  \label{L1}
  L_1(Z)&=&\sum_{i,j=1}^{n}{\ln(1+|X_{i}-Y_{j}|)},\\
  \label{L2}
  L_2(Z)&=&\sum_{i,j=1}^{n}{\ln(1+|X_{i}-Y_{j}|^2)},\\
  \label{Linf}
  L_\infty(Z)&=&\sum_{i,j=1}^{n}{\ln(|X_{i}-Y_{j}|)},\\
  \label{LL}
  LL_{distribution} &=& \text{maximum log likelyhood permutation criterion based on the distribution}
\end{eqnarray}
